package linkedlist;
import java.util.*;

import linkedlist.singleylistinsertion.SinglyLinkedList;


public class inserting {
	public static void main(String[] args) {
        SinglyLinkedList list = new SinglyLinkedList();
    	Scanner sc =new Scanner(System.in);

        list.add(10);
        list.add(20);
        list.add(30);
        list.add(40);
        list.add(50);
        
        System.out.println("linkedlist: " );
        list.printList();


System.out.println("Enter the number to be inserted in the begining: ");
int a=sc.nextInt();

list.insertAtBeginning(a);
list.printList();

System.out.println("Enter the number to be inserted  ");

int b=sc.nextInt();
System.out.println("enter the position :");
int pos=sc.nextInt();


list.insertAtPosition(b,pos);

System.out.print("Updated Singly LinkedList: " );
list.printList();
}}
